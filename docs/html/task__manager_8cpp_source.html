189JsonResult TaskManager::fromJsonString(const std::string& json_str) {
190 try {
191 // Simple JSON parsing - extract next_id and tasks array
192 auto findValue = [&json_str](const std::string& key) -> std::string {
193 std::string search_key = "\"" + key + "\":";
194 size_t pos = json_str.find(search_key);
195 if (pos == std::string::npos) return "";
196
197 pos += search_key.length();
198 while (pos < json_str.length() && std::isspace(json_str[pos])) ++pos;
199
200 if (pos >= json_str.length()) return "";
201
202 if (json_str[pos] == '"') {
203 // String value
204 ++pos;
205 size_t end_pos = pos;
206 while (end_pos < json_str.length() && json_str[end_pos] != '"') {
207 if (json_str[end_pos] == '\\') ++end_pos; // Skip escaped characters
208 ++end_pos;
209 }
210 return json_str.substr(pos, end_pos - pos);
211 } else {
212 // Number or other value
213 size_t end_pos = pos;
214 while (end_pos < json_str.length() &&
215 json_str[end_pos] != ',' &&
216 json_str[end_pos] != '}' &&
217 json_str[end_pos] != '\n' &&
218 json_str[end_pos] != '\r') {
219 ++end_pos;
220 }
221 std::string value = json_str.substr(pos, end_pos - pos);
222 // Trim whitespace
223 value.erase(0, value.find_first_not_of(" \t\n\r"));
224 value.erase(value.find_last_not_of(" \t\n\r") + 1);
225 return value;
226 }
227 };
228
229 std::string next_id_str = findValue("next_id");
230 if (!next_id_str.empty()) {
231 _next_id = std::stoi(next_id_str);
232 }
233
234 // Find tasks array
235 size_t tasks_pos = json_str.find("\"tasks\":");
236 if (tasks_pos == std::string::npos) {
237 return std::unexpected(JsonError::InvalidFormat);
238 }
239
240 size_t array_start = json_str.find("[", tasks_pos);
241 if (array_start == std::string::npos) {
242 return std::unexpected(JsonError::InvalidFormat);
243 }
244
245 size_t array_end = json_str.find("]", array_start);
246 if (array_end == std::string::npos) {
247 return std::unexpected(JsonError::InvalidFormat);
248 }
249
250 // Parse individual task objects
251 _tasks.clear();
252 size_t pos = array_start + 1;
253
254 while (pos < array_end) {
255 // Find next task object
256 size_t obj_start = json_str.find("{", pos);
257 if (obj_start >= array_end) break;
258
259 // Find matching closing brace
260 size_t obj_end = obj_start + 1;
261 int brace_count = 1;
262 while (obj_end < array_end && brace_count > 0) {
263 if (json_str[obj_end] == '{') ++brace_count;
264 else if (json_str[obj_end] == '}') --brace_count;
265 ++obj_end;
266 }
267
268 if (brace_count == 0) {
269 std::string task_json = json_str.substr(obj_start, obj_end - obj_start);
270 auto task_result = Task::fromJson(task_json);
271 if (task_result) {
272 _tasks.push_back(*task_result);
273 } else {
274 return std::unexpected(task_result.error());
275 }
276 }
277
278 pos = obj_end;
279 }
280
281 return true;
282 } catch (const std::exception&) {
283 return std::unexpected(JsonError::ParseError);
284 }
