577void App::displayJsonAsTable(const std::string& json_content) {
578 // Simple parsing to extract basic info
579 auto findValue = [&json_content](const std::string& key) -> std::string {
580 std::string search_key = "\"" + key + "\":";
581 size_t pos = json_content.find(search_key);
582 if (pos == std::string::npos) return "";
583
584 pos += search_key.length();
585 while (pos < json_content.length() && std::isspace(json_content[pos])) ++pos;
586
587 if (pos >= json_content.length()) return "";
588
589 if (json_content[pos] == '"') {
590 ++pos;
591 size_t end_pos = pos;
592 while (end_pos < json_content.length() && json_content[end_pos] != '"') {
593 if (json_content[end_pos] == '\\') ++end_pos;
594 ++end_pos;
595 }
596 return unescapeJsonString(json_content.substr(pos, end_pos - pos));
597 } else {
598 size_t end_pos = pos;
599 while (end_pos < json_content.length() &&
600 json_content[end_pos] != ',' &&
601 json_content[end_pos] != '}' &&
602 json_content[end_pos] != '\n' &&
603 json_content[end_pos] != '\r') {
604 ++end_pos;
605 }
606 std::string value = json_content.substr(pos, end_pos - pos);
607 value.erase(0, value.find_first_not_of(" \t\n\r"));
608 value.erase(value.find_last_not_of(" \t\n\r") + 1);

664std::vector<App::TaskInfo> App::parseTasksFromJson(const std::string& json_content) {
665 std::vector<TaskInfo> tasks;
666
667 // Find tasks array
668 size_t tasks_pos = json_content.find("\"tasks\":");
669 if (tasks_pos == std::string::npos) {
670 return tasks; // No tasks array found
671 }
672
673 // Find opening bracket of tasks array
674 size_t array_start = json_content.find('[', tasks_pos);
675 if (array_start == std::string::npos) {
676 return tasks;
677 }
678
679 // Simple parser to extract task objects
680 size_t pos = array_start + 1;
681 while (pos < json_content.length()) {
682 // Skip whitespace
683 while (pos < json_content.length() && std::isspace(json_content[pos])) {
684 pos++;
685 }
686
687 if (pos >= json_content.length() || json_content[pos] == ']') {
688 break; // End of array
689 }
690
691 if (json_content[pos] == '{') {
692 // Parse task object
693 TaskInfo task;
694 task.id = 0;
695 task.priority = 0;
696
697 size_t obj_end = pos;
698 int brace_count = 1;
699 obj_end++;
700
701 // Find end of object
702 while (obj_end < json_content.length() && brace_count > 0) {
703 if (json_content[obj_end] == '{') brace_count++;
704 else if (json_content[obj_end] == '}') brace_count--;
705 obj_end++;
706 }
707
708 std::string task_obj = json_content.substr(pos, obj_end - pos);
709
710 // Extract task properties
711 auto extractValue = [&task_obj](const std::string& key) -> std::string {
712 std::string search_key = "\"" + key + "\":";
713 size_t key_pos = task_obj.find(search_key);
714 if (key_pos == std::string::npos) return "";
715
716 key_pos += search_key.length();
717 while (key_pos < task_obj.length() && std::isspace(task_obj[key_pos])) {
718 key_pos++;
719 }
720
721 if (key_pos >= task_obj.length()) return "";
722
723 if (task_obj[key_pos] == '"') {
724 key_pos++;
725 size_t end_pos = key_pos;
726 while (end_pos < task_obj.length() && task_obj[end_pos] != '"') {
727 if (task_obj[end_pos] == '\\') end_pos++;
728 end_pos++;
729 }
730 return unescapeJsonString(task_obj.substr(key_pos, end_pos - key_pos));
731 } else {
732 size_t end_pos = key_pos;
733 while (end_pos < task_obj.length() &&
734 task_obj[end_pos] != ',' &&
735 task_obj[end_pos] != '}' &&
736 task_obj[end_pos] != '\n' &&
737 task_obj[end_pos] != '\r') {
738 end_pos++;
739 }
740 std::string value = task_obj.substr(key_pos, end_pos - key_pos);
741 value.erase(0, value.find_first_not_of(" \t\n\r"));
742 value.erase(value.find_last_not_of(" \t\n\r") + 1);
743 return value;
744 }
745 };
746
747 // Parse task fields
748 std::string id_str = extractValue("id");
749 if (!id_str.empty()) {
750 task.id = std::stoi(id_str);
751 }
752
753 task.title = extractValue("title");
754 task.status = extractValue("status");
755 task.category = extractValue("category");
756 task.created_at = extractValue("created_at");
757 task.description = extractValue("description");
758
759 std::string priority_str = extractValue("priority");
760 if (!priority_str.empty()) {
761 task.priority = std::stoi(priority_str);
762 }
763
764 tasks.push_back(task);
765 pos = obj_end;
766 } else {
767 pos++;
768 }
769 }
770
771 return tasks;
772}
