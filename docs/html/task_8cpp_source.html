168/**
169 * @brief Escape special characters in a string for JSON encoding
170 * @param str The input string to escape
171 * @return JSON-safe escaped string
172 * @details Handles quotes, backslashes, control characters, and UTF-8
173 */
174std::string escapeJsonString(const std::string& str) {
175 std::string escaped;
176 escaped.reserve(str.size() + 20); // Reserve some extra space for escapes
177
178 for (unsigned char c : str) {
179 switch (c) {
180 case '"': escaped += "\\\""; break;
181 case '\\': escaped += "\\\\"; break;
182 case '\b': escaped += "\\b"; break;
183 case '\f': escaped += "\\f"; break;
184 case '\n': escaped += "\\n"; break;
185 case '\r': escaped += "\\r"; break;
186 case '\t': escaped += "\\t"; break;
187 default:
188 // Only escape actual control characters (0-31), not UTF-8 bytes
189 if (c < 32) {
190 escaped += std::format("\\u{:04x}", c);
191 } else {
192 // For UTF-8 and normal ASCII (32-255), keep as-is
193 escaped += static_cast<char>(c);
194 }
195 break;
196 }
197 }
198 return escaped;
199}

201/**
202 * @brief Unescape JSON string by converting escape sequences to characters
203 * @param str The JSON string to unescape
204 * @return Unescaped string
205 */
206std::string unescapeJsonString(const std::string& str) {
207 std::string unescaped;
208 unescaped.reserve(str.size());
209
210 for (size_t i = 0; i < str.size(); ++i) {
211 if (str[i] == '\\' && i + 1 < str.size()) {
212 switch (str[i + 1]) {
213 case '"': unescaped += '"'; ++i; break;
214 case '\\': unescaped += '\\'; ++i; break;
215 case 'b': unescaped += '\b'; ++i; break;
216 case 'f': unescaped += '\f'; ++i; break;
217 case 'n': unescaped += '\n'; ++i; break;
218 case 'r': unescaped += '\r'; ++i; break;
219 case 't': unescaped += '\t'; ++i; break;
220 default: unescaped += str[i]; break;
221 }
222 } else {
223 unescaped += str[i];
224 }
225 }
226 return unescaped;
227}

232 * @return ISO 8601 formatted string (YYYY-MM-DDTHH:MM:SS.mmm)

234std::string timePointToIsoString(const std::chrono::system_clock::time_point& tp) {
235 auto time_t = std::chrono::system_clock::to_time_t(tp);
236 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(tp.time_since_epoch()) % 1000;
237
238 std::ostringstream oss;
239 // Fix: Use localtime instead of gmtime to show local timezone
240 oss << std::put_time(std::localtime(&time_t), "%Y-%m-%dT%H:%M:%S");
241 oss << std::format(".{:03d}", ms.count()); // Remove Z suffix for local time
242 return oss.str();
243}

245/**
246 * @brief Parse ISO 8601 formatted string to time_point
247 * @param iso_str The ISO 8601 string to parse
248 * @return Corresponding time_point
249 * @details Handles millisecond precision if present
250 */
251std::chrono::system_clock::time_point isoStringToTimePoint(const std::string& iso_str) {
252 std::tm tm = {};
253 std::istringstream ss(iso_str);
254 ss >> std::get_time(&tm, "%Y-%m-%dT%H:%M:%S");
255
256 auto tp = std::chrono::system_clock::from_time_t(std::mktime(&tm));
257
258 // Handle milliseconds if present
259 auto dot_pos = iso_str.find('.');
260 if (dot_pos != std::string::npos) {
261 auto ms_str = iso_str.substr(dot_pos + 1, 3);
262 if (!ms_str.empty() && std::isdigit(ms_str[0])) {
263 int ms = std::stoi(ms_str);
264 tp += std::chrono::milliseconds(ms);
265 }
266 }
267
268 return tp;
269}

296/**
297 * @brief Create a Task from a JSON string
298 * @param json_str JSON string to parse
299 * @return Task object if successful, or JsonError if parsing failed
300 * @details C++23 Feature: Uses std::expected for error handling
301 */
302std::expected<Task, JsonError> Task::fromJson(const std::string& json_str) {
303 try {
304 // Simple JSON parsing - in production, use a proper JSON library
305 auto findValue = [&json_str](const std::string& key) -> std::string {
306 std::string search_key = "\"" + key + "\":";
307 size_t pos = json_str.find(search_key);
308 if (pos == std::string::npos) return "";
309
310 pos += search_key.length();
311 while (pos < json_str.length() && std::isspace(json_str[pos])) ++pos;
312
313 if (pos >= json_str.length()) return "";
314
315 if (json_str[pos] == '"') {
316 // String value
317 ++pos;
318 size_t end_pos = pos;
319 while (end_pos < json_str.length() && json_str[end_pos] != '"') {
320 if (json_str[end_pos] == '\\') ++end_pos; // Skip escaped characters
321 ++end_pos;
322 }
323 return unescapeJsonString(json_str.substr(pos, end_pos - pos));
324 } else {
325 // Number or other value
326 size_t end_pos = pos;
327 while (end_pos < json_str.length() &&
328 json_str[end_pos] != ',' &&
329 json_str[end_pos] != '}' &&
330 json_str[end_pos] != '\n' &&
331 json_str[end_pos] != '\r') {
332 ++end_pos;
333 }
334 std::string value = json_str.substr(pos, end_pos - pos);
335 // Trim whitespace
336 value.erase(0, value.find_first_not_of(" \t\n\r"));
337 value.erase(value.find_last_not_of(" \t\n\r") + 1);
338 return value;
339 }
340 };
341
342 std::string id_str = findValue("id");
343 std::string title = findValue("title");
344 std::string description = findValue("description");
345 std::string status_str = findValue("status");
346 std::string category = findValue("category");
347 std::string priority_str = findValue("priority");
348 std::string created_at_str = findValue("created_at");
349 std::string updated_at_str = findValue("updated_at");
350 std::string completed_at_str = findValue("completed_at");
351
352 if (id_str.empty() || title.empty()) {
353 return std::unexpected(JsonError::InvalidFormat);
354 }
355
356 int id = std::stoi(id_str);
357 auto status_opt = stringToTaskStatus(status_str);
358 if (!status_opt) {
359 return std::unexpected(JsonError::InvalidFormat);
360 }
361
362 Task task(id, title, description, *status_opt);
363
364 if (!category.empty()) {
365 task.setCategory(category);
366 }
367
368 if (!priority_str.empty()) {
369 int priority = std::stoi(priority_str);
370 task.setPriority(priority);
371 }
372
373 if (!created_at_str.empty()) {
374 task._metadata.created_at = isoStringToTimePoint(created_at_str);
375 }
376
377 if (!updated_at_str.empty()) {
378 task._metadata.updated_at = isoStringToTimePoint(updated_at_str);
379 }
380
381 if (!completed_at_str.empty()) {
382 task._metadata.completed_at = isoStringToTimePoint(completed_at_str);
383 }
384
385 return task;
386 } catch (const std::exception&) {
387 return std::unexpected(JsonError::ParseError);
388 }
389}
